132 static void UART_SetBaudRate(const u32 baudRate, const UART_MODE_t mode, volatile u8 * const UBRRnH, volatile u8 * const UBRRnL) {
133  u16 ubbrValue = 0;
134 
135  /* Calculate UBRR value */
136  switch (mode) {
137  case UART_MODE_ASYNCHRONOUS_NORMAL:
138  ubbrValue = (u16)( ( ( (u32)F_CPU ) / ((u32)16 * baudRate) ) - 1);
139  break;
140  case UART_MODE_ASYNCHRONOUS_DOUBLE_SPEED:
141  ubbrValue = (u16)( ( ( (u32)F_CPU ) / ((u32)8 * baudRate) ) - 1);
142  break;
143  case UART_MODE_SYNCHRONOUS_MASTER:
144  ubbrValue = (u16)( ( ( (u32)F_CPU ) / ((u32)2 * baudRate) ) - 1);
145  break;
146  default:
147  break;
148  }
149 
150  /* Set UBRR value. UBRRH must be written before UBRRL */
151  *UBRRnH = (u8)(ubbrValue >> 8);
152  *UBRRnL = (u8)ubbrValue;
153 }
