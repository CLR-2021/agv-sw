242 ERROR_STATUS_t SPI_SendByte(const u8 data) {
243  ERROR_STATUS_t error = ERROR_NO;
244  u8 volatile fBuffer = 0;
245 
246  SPDR = data; /* Write data to the SPI data register */
247 
248  while(BIT_IS_CLEAR(SPSR, SPIF))
249  ; /* Wait until SPI transfer complete */
250 
251  /* Clear transmission complete flag by reading SPI Status register */
252  fBuffer = SPSR; /* Read SPI Status register */
253  fBuffer = SPDR; /* Then access SPI Data register */
254 
255  return error;
